# 小实验

一个小实验，可能只需要一个文件，数据集、模型、训练算法写在一起，很多这样的实验都会有五六百行，而且是通过传递训练方法的方式来进行训练，这种方式很不利于定位核心代码，每定位一次变量的定义可能都要定位好几次，而这在方法名相同的情况下就更难受了。

下面展示一个利用 thexp 完成的，代码结构良好的MNIST训练过程，这是接下来其他示例的基础：

from torch.nn import functional as F
import torch.nn as nn
import torch
from thexp import *

# glob.add_value('datasets', '/home/share/yanghaozhe/pytorchdataset', glob.LEVEL.repository)

params = Params()
params.device = 'cuda:1'
params.epoch = 5
params.batch_size = 128
params.topk = (1, 4)
params.from_args()
# params.root = glob['datasets']
params.root = '/home/share/yanghaozhe/pytorchdataset'
params.dataloader = dict(shuffle=True, batch_size=32, drop_last=True)
params.optim = dict(lr=0.01, weight_decay=0.09, momentum=0.9)


def MNIST(mode="train", download=False):
    # 由于不同数据集区分训练或测试集的方式不同，因此建议数据集以统一的接口定义
    from torchvision.datasets.mnist import MNIST
    from torchvision import transforms
    weak = transforms.ToTensor()
    if mode == "train":
        return MNIST(root=params.root, train=True, download=download, transform=weak)
    else:
        return MNIST(root=params.root, train=False, transform=weak, download=download)


class SimpleNet(nn.Module):
    """模型定义部分"""

    def __init__(self):
        super(SimpleNet, self).__init__()
        self.convnet = nn.Sequential(nn.Conv2d(1, 32, 5), nn.PReLU(),
                                     nn.MaxPool2d(2, stride=2),
                                     nn.Conv2d(32, 64, 5), nn.PReLU(),
                                     nn.MaxPool2d(2, stride=2))

        self.fc = nn.Sequential(nn.Linear(64 * 4 * 4, 256),
                                nn.Tanh(),
                                nn.Linear(256, 128),
                                nn.Tanh(),
                                nn.Linear(128, 10)
                                )

    def forward(self, x):
        output = self.convnet(x)
        output = output.view(output.size()[0], -1)
        output = self.fc(output)
        return output


class MNISTTrainer(Trainer):
    __exp_name__ = "MNISTDemo"

    def callbacks(self, params: Params):
        callbacks.LoggerCallback().hook(self)
        callbacks.AutoRecord().hook(self)
        callbacks.EvalCallback(test_per_epoch=5).hook(self)

    def datasets(self, params: Params):
        from torch.utils.data.dataloader import DataLoader
        from thexp.torch.data.collate import AutoCollate

        # self.device = torch.device(param.device)
        train_loader = DataLoader(MNIST(mode='train'), **params.dataloader, collate_fn=AutoCollate(self.device))
        test_loader = DataLoader(MNIST(mode='test'), **params.dataloader, collate_fn=AutoCollate(self.device))

        self.regist_databundler(
            train=train_loader,
            test=test_loader,
        )

    def models(self, params: Params):
        from torch.optim import SGD
        self.model = SimpleNet().to(self.device)
        self.optim = SGD(self.model.parameters(), **params.optim)

    def train_batch(self, eidx, idx, global_step, batch_data, params, device):
        optim = self.optim # 你可以直接调用 self.optim，也可以用这种方式，看你喜欢哪一种
        meter = Meter()
        xs, ys = batch_data

        # 训练逻辑
        logits = self.model(xs)
        meter.loss = F.cross_entropy(logits, ys)
        self.train_precise(logits, ys, meter)
        # 反向传播
        meter.loss.backward()
        optim.step()
        optim.zero_grad()

        return meter

    def train_precise(self, logits, labels, meter):
        """train batch accuracy"""
        with torch.no_grad():
            _, maxid = torch.max(logits, dim=-1)
            total = labels.size(0)
            top1 = (labels == maxid).sum().item()
            meter.acc = top1 / total
            meter.percent(meter.acc_)
        return meter.acc

    def test_eval_logic(self, dataloader, param: Params):
        from thexp.calculate import accuracy as acc
        with torch.no_grad():
            count_dict = Meter()
            for xs, labels in dataloader:
                xs, labels = xs.to(self.device), labels.to(self.device)
                preds = self.predict(xs)
                total, topk_res = acc.classify(preds, labels, topk=param.topk)
                count_dict["total"] += total
                for i, topi_res in zip(param.topk, topk_res):
                    count_dict[i] += topi_res
        return count_dict

    def predict(self, xs):
        return self.model(xs)


# params
for p in params.grid_search("optim.lr", [0.001, 0.005, 0.0005]):
    for pp in p.grid_search("epoch", [10, 15, 20]):
        trainer = MNISTTrainer(pp)
        trainer.train()

# 将变化拆分到各文件

对多变的部份，建议将其额外拆分到一个单文件中。

# 扩展数据集

如果你除了要训练MNIST，也要训练FashionMNIST，那么就建议你将两个数据集的读取部份单独放在一个文件里：

from thexp import glob

root = glob['datasets']

def MNIST(mode="train", download=False):
    # 由于不同数据集区分训练或测试集的方式不同，因此建议数据集以统一的接口定义
    from torchvision.datasets.mnist import MNIST
    from torchvision import transforms
    weak = transforms.ToTensor()
    if mode == "train":
        return MNIST(root=root, train=True, download=download, transform=weak)
    else:
        return MNIST(root=root, train=False, transform=weak, download=download)


def FMNIST(mode="train", download=False):
    # 由于不同数据集区分训练或测试集的方式不同，因此建议数据集以统一的接口定义
    from torchvision.datasets.mnist import FashionMNIST
    from torchvision import transforms
    weak = transforms.ToTensor()
    if mode == "train":
        return FashionMNIST(root=root, train=True, download=download, transform=weak)
    else:
        return FashionMNIST(root=root, train=False, transform=weak, download=download)

    def datasets(self, params: Params):
        from torch.utils.data.dataloader import DataLoader
        from thexp.torch.data.collate import AutoCollate

        else:
            assert False

        self.regist_databundler(
            train=train_loader,
            test=test_loader,
        )

        ...

        if params.ema:
            from tricks.ema import ema
            self.ema_model = ema(self.model)

        ...
    ...
    def on_train_batch_end(self, trainer: 'BaseTrainer', func, params: Params, meter: Meter, *args, **kwargs):
        if params.ema:
            from tricks.ema import update_ema_variables
            update_ema_variables(self.model, self.ema_model, params.global_step)
    ...
    def predict(self, xs):
        if params.ema:
            return self.ema_model(xs)
        else:
            return self.model(xs)

# 什么时候需要多个Trainer？

通过之前的过程我们可以看到，当数据集或模型结构有变化的时候，我们只需要将其变化拆分到其他文件中，随后在参数中添加选项，就可以保持良好的代码结构，这个时候即使只有一个 Trainer 也可以，如果让每个 Trainer 对应不同的数据集或模型结构，反而会导致代码的冗余。

那么什么时候需要多个 Trainer 呢？深度学习三件套，数据、模型、算法，当算法改变，也即train_batch()方法中的逻辑都要修改时，你可能就需要考虑使用多个 Trainer 了。

比如说，在交叉熵损失函数的基础上，你想要添加三元组损失函数，此时数据的加载方式肯定会大幅改变，由 (xs,ys) 转变成 (xs,pos,neg,ys)。对于数据集的变化，你可以通过在相应的文件夹中添加 triplet 文件或文件夹（这取决于你要改动的数据集数量）。而在 Trainer 中，你会有两个选择：

  • 通过在 params 中添加可选项，在 train_batch() 逻辑中进行区分
  • 继承创建另外一个 Trainer （这个时候，可能需要单独创建一个trainer文件存放每一种Trainer），用于训练添加了三元组损失函数的模型

这两种方法没有优劣之分，决定采用哪一种选择的关键在于，你认为你新添加的这一方法，是另一个实验，还是该实验的一个对比实验？thexp 在对目录组织的分级中明确了项目和实验的关系（一个项目可以对应多个实验），而在Trainer中，也存在一个类变量 __exp_name__ 用来标识该 Trainer 的实验名。因此，是否拆分 Trainer ，由你这项更改的性质决定。

TIP

比如，一个sota的代码中会存在对之前其他论文代码的复现实验，这明显属于不同的实验，因此多个 Trainer 肯定是更好的选择。

当 Trainer 被拆分的时候，出于文件组织的考虑，运行主函数就不推荐写在 Trainer 定义的文件中。此时，将主运行文件放在项目根目录下，一个 Trainer（也是一个实验）对应一个运行文件，可能会更好一些。在这种情况下，数据、模型、算法、运行被很好的区分开，保证了代码结构尽可能的扁平化，一处的更改不会引起其他文件的大幅度更改。

# 在全局使用 params

实际上，如果你控制好引用关系避免递归引用，你完全可以将 params 变成全局的参数。虽然exp 中也提供了运行期间的全局参数（再次觉得这可能是个鸡肋），但 params 在项目中定义的变量将是 IDE 友好型的，能够在全局被自动补全检测到。

Params放在哪？

当你将 params 作为全局参数的时候，比较推荐单独将其放在一个文件中，命名为 config.py 或 params.py 之类的。随后在各个文件中引入并使用。

# 是否有必要定义多个 Params

当定义了多个 Trainer 的时候，很有必要考虑一下是否有必要为每个 Trainer 定义不同 Params。不断的往一个 Params 中添加参数固然统一，但却有可能导致每个实验会保存大量的冗余参数（只是有可能）。

一个可选项是利用继承属性，在基类中定义共有的变量，在子类中定义特有的变量。但这种方式，没有办法很好的将params作为全局变量。

一个还算推荐的写法是仍然将全局参数写在 config.py 中，随后在各个 Trainer 文件中引入并添加特有的变量。这样可以比较好的保证各个 Trainer 之间的参数不会互相污染。

使用权和修改权

Trainer 最好和文件是一一对应的关系，没有必要在一个文件中包含多个 Trainer。基于此，Params 应该只在定义文件 config.py 和 各个 Trainer 文件中进行修改，在其他文件中，应该只对其进行读取，不应该对其进行修改。

# 扩展thexp功能

你可能想对thexp 提供的回调功能进行扩展，或想添加 thexp 中提供的某些分类的功能。此外，由于 thexp 缺少测试，那么或许对于其中的一些功能，你使用的时候会出现一些小bug，你会在本地新建一个同名文件来修改好这个bug。

在这种情况下，我是统一将其放在 thextra 文件夹下。其作用是提醒我：在结束这个实验后，记得把这个文件夹里的东西添加到 thexp 上去。
